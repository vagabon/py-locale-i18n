# Create a translator object
translator = Translator()

# Define the JSON object in French
json_object_fr = {
  "TITLE": "News",
  "UPDATE": "Mise  jour d'une news",
  "CREATE": "Cration d'une news",
  "FIELDS": {
    "TITLE": "Titre",
    "RESUME": "Rsum",
    "DESCRIPTION": "Description",
    "AVATAR": "Avatar",
    "IMAGE": "Image",
    "TAGS": "tiquettes",
    "USER": "Utilisateur"
  }
}

# Translate the JSON object to English
json_object_en = {}
for key, value in json_object_fr.items():
    if isinstance(value, dict):
        # If the value is a nested dictionary, translate its values recursively
        json_object_en[key] = {}
        for nested_key, nested_value in value.items():
            translated_value = translator.translate(nested_value, src="fr", dest="en").text
            json_object_en[key][nested_key] = translated_value
    else:
        # If the value is a string, translate it
        translated_value = translator.translate(value, src="fr", dest="en").text
        json_object_en[key] = translated_value

# Print the translated JSON object
print(json_object_en)

We then iterate over the key-value pairs in the JSON object and translate the values to English using the translate method of the Translator object. If the value is a nested dictionary, we translate its values recursively.
